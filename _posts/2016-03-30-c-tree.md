#include<stdio.h>

struct Node{
	int value;
	Node * left, * right;
};

class Tree{
private:

	Node * root;
	int * data;

	Node * CreateTree(){
		data++;
		if(*data == 0){
			return NULL;
		}

		Node * node = new Node();
		
		node->value = *(data);
		node->left = CreateTree();
		node->right = CreateTree();
		
		return node;
	}

	void Delete(Node * node){
		if(node == NULL){
			return;
		}
		Delete(node->left);
		Delete(node->right);
		delete node;
		node = NULL;
	}
	
	void PrintPreOrder(Node * node){
		if(node == NULL){
			printf("%d ", 0);
			return;
		}
		printf("%d ", node->value);
		PrintPreOrder(node->left);
		PrintPreOrder(node->right);
	}
	
	void PrintInOrder(Node * node){
		if(node == NULL){
			printf("%d ", 0);
			return;
		}
		PrintInOrder(node->left);
		printf("%d ", node->value);
		PrintInOrder(node->right);
	}
	
	void PrintPosOrder(Node * node){
		if(node == NULL){
			printf("%d ", 0);
			return;
		}
		PrintPosOrder(node->left);
		PrintPosOrder(node->right);
		printf("%d ", node->value);
	}

public:
	
	Tree(){
		root = NULL;
	}
	
	~Tree(){
		Clear();
	}

	void Create(int * arr){
		if(arr == NULL){
			return ;
		}
		if(root != NULL){
			Clear();
		}
		data = arr - 1;
		root = CreateTree();
	}

	void Clear(){
		Delete(root);
		root = NULL;//need
	}

	void Print(int order){
		switch(order){
		case 0:
			{
				PrintPreOrder(root);
			}
			break;
		case 1:
			{
				PrintInOrder(root);
			}
			break;
		case 2:
			{
				PrintPosOrder(root);
			}
			break;
		}
		printf("\n");
	}
};

int main(){
	int arr[] = {1, 2, 3, 0, 0, 0, 4, 0, 0};
	Tree * t = new Tree();
	t->Create(arr);
	t->Print(0);
	return 0;
}
